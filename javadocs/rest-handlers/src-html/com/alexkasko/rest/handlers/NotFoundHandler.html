001    package com.alexkasko.rest.handlers;
002    
003    import javax.servlet.http.HttpServletRequest;
004    import javax.servlet.http.HttpServletResponse;
005    
006    /**
007     * Used when no handler can be found for provided request
008     *
009     * @author alexkasko
010     * Date: 11/14/12
011     */
012    public interface NotFoundHandler {
013        /**
014         * Implementation may send 404 to client
015         *
016         * @param dispatcher handlers dispatcher
017         * @param request request
018         * @param response response
019         * @throws Exception on any exception
020         */
021        void handle(HandlersDispatcher dispatcher, HttpServletRequest request, HttpServletResponse response) throws Exception;
022    }
