001    package com.alexkasko.rest.handlers;
002    
003    import javax.servlet.http.HttpServletRequest;
004    import javax.servlet.http.HttpServletResponse;
005    import java.util.Map;
006    
007    /**
008     * Transport level request handling.
009     * Should process request transport (application level transport, not TCP) and
010     * delegate application processing to {@link RestHandler}
011     *
012     * @author alexey
013     * Date: 8/9/12
014     * @see RestHandler
015     */
016    
017    public interface TransportHandler<T extends RestHandler> {
018        /**
019         * Processes request transport (application level transport, not TCP) and
020         * delegate application processing to {@link RestHandler}
021         *
022         * @param handlerClass application handler class
023         * @param request request
024         * @param response response
025         * @param urlParams named parameters from request path
026         * @throws Exception on any application exception
027         */
028        void handle(Class<? extends T> handlerClass, HttpServletRequest request, HttpServletResponse response,
029                    Map<String, String> urlParams) throws Exception;
030    }
