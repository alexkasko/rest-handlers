001    package com.alexkasko.rest.handlers;
002    
003    import com.google.code.regexp.NamedMatcher;
004    import com.google.code.regexp.NamedPattern;
005    
006    /**
007     * Handler URL mapping entry, used in {@link com.alexkasko.rest.handlers.HandlersDispatcher#builder()}.
008     * Made public for mappings introspection in {@link NotFoundHandler}
009     *
010     * @author alexkasko
011     * Date: 11/14/12
012     * @see HandlersDispatcher.Builder
013     */
014    public class HandlersMappingEntry<T extends RestHandler> {
015        private final NamedPattern pattern;
016        private final TransportHandler<T> ra;
017        private final Class<T> clazz;
018    
019        /**
020         * Package private constructor
021         *
022         * @param pattern named regex pattern
023         * @param th transport handler
024         * @param clazz app handler class
025         */
026        HandlersMappingEntry(String pattern, TransportHandler<T> th, Class<T> clazz) {
027            if(null == pattern) throw new IllegalArgumentException("Provided pattern is null");
028            if(null == th) throw new IllegalArgumentException("Provided transport handler is null");
029            if(null == clazz) throw new IllegalArgumentException("Provided handler class is null");
030            this.pattern = NamedPattern.compile(pattern);
031            this.ra = th;
032            this.clazz = clazz;
033        }
034    
035        /**
036         * Matched request path info
037         *
038         * @param input request path info
039         * @return matcher
040         */
041        NamedMatcher matcher(String input) {
042            return pattern.matcher(input);
043        }
044    
045        /**
046         * Returns named regex pattern string
047         *
048         * @return named regex pattern as it was provided to {@link HandlersDispatcher.Builder}
049         */
050        public String getPattern() {
051            return pattern.namedPattern();
052        }
053    
054        /**
055         * Returns transport handler
056         *
057         * @return transport handler
058         */
059        public TransportHandler<T> getTransportHandler() {
060            return ra;
061        }
062    
063        /**
064         * Returns app handler class
065         *
066         * @return handler class
067         */
068        public Class<T> getRestHandlerClass() {
069            return clazz;
070        }
071    }
