001package com.alexkasko.rest.handlers;
002
003import javax.servlet.ServletException;
004import javax.servlet.http.HttpServlet;
005import javax.servlet.http.HttpServletRequest;
006import javax.servlet.http.HttpServletResponse;
007import java.io.IOException;
008import java.io.UnsupportedEncodingException;
009
010/**
011 * Dispatcher servlet for handlers, delegates {@code GET}, {@code POST} and {@code PUT}
012 * dispatching to {@link HandlersDispatcher}. Accesses dispatcher through {@link javax.servlet.ServletContext}
013 * using mandatory init parameter {@code dispatcherKeyInServletContext}.
014 * Uses {@code UTF-8} as default encoding for request and response.
015 *
016 * @author  alexkasko
017 * Date: 3/18/12
018 * @see HandlersDispatcher
019 */
020public class HandlersDispatcherServlet extends HttpServlet {
021    private static final long serialVersionUID = 7041995029366447857L;
022    private String dispatcherKey;
023
024    /**
025     * Checks availability of {@code dispatcherKeyInServletContext} init parameter
026     *
027     * @throws ServletException on missed init parameter
028     */
029    @Override
030    public void init() throws ServletException {
031        String key = this.getServletConfig().getInitParameter("dispatcherKeyInServletContext");
032        if(null == key) throw new ServletException("Servlet init parameter 'dispatcherKeyInServletContext' " +
033                "must be provided. It will be using for accessing 'HandlersDispatcher' in 'ServletContext'");
034        this.dispatcherKey = key;
035    }
036
037    /**
038     * Delegates request dispatching to {@link HandlersDispatcher}
039     *
040     * @param req request
041     * @param resp response
042     * @throws ServletException on {@link HandlersDispatcher} not found in ServletContext
043     * @throws IOException cannot happen
044     */
045    @Override
046    public void doGet (HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
047        dispatch(req, resp);
048    }
049
050    /**
051     * Delegates request dispatching to {@link HandlersDispatcher}
052     *
053     * @param req request
054     * @param resp response
055     * @throws ServletException on {@link HandlersDispatcher} not found in ServletContext
056     * @throws IOException cannot happen
057     */
058    @Override
059    public void doPost (HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
060        dispatch(req, resp);
061    }
062
063    /**
064     * Delegates request dispatching to {@link HandlersDispatcher}
065     *
066     * @param req request
067     * @param resp response
068     * @throws ServletException on {@link HandlersDispatcher} not found in ServletContext
069     * @throws IOException cannot happen
070     */
071    @Override
072    protected void doPut(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
073        dispatch(req, resp);
074    }
075
076    /**
077     * Delegates request dispatching to {@link HandlersDispatcher}
078     *
079     * @param req request
080     * @param resp response
081     * @throws ServletException on {@link HandlersDispatcher} not found in ServletContext
082     * @throws IOException cannot happen
083     */
084    @Override
085    protected void doDelete(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
086        dispatch(req, resp);
087    }
088
089    private void dispatch(HttpServletRequest req, HttpServletResponse resp) throws UnsupportedEncodingException, ServletException {
090        req.setCharacterEncoding("UTF-8");
091        resp.setCharacterEncoding("UTF-8");
092        dispatcher().dispatch(req, resp);
093    }
094
095    private HandlersDispatcher dispatcher() throws ServletException {
096        HandlersDispatcher disp = (HandlersDispatcher) getServletContext().getAttribute(dispatcherKey);
097        if(null == disp) throw new ServletException("Cannot find dispatcher in ServletContext using key: '"
098                + dispatcherKey + "' check 'dispatcherKeyInServletContext' init parameter for 'HandlersDispatcherServlet'");
099        return disp;
100    }
101}
