001    package com.alexkasko.rest.handlers;
002    
003    import com.google.code.regexp.NamedMatcher;
004    
005    import javax.servlet.http.HttpServletRequest;
006    import javax.servlet.http.HttpServletResponse;
007    import java.util.ArrayList;
008    import java.util.List;
009    
010    import static java.util.Collections.unmodifiableList;
011    
012    /**
013     * Dispatches request using handlers registry matching with regular expression.
014     * Contains separate registries for {@code GET}, {@code POST} and {@code PUT} requests.
015     * Registries are ordered in handlers registration order. First matched handler will be used.
016     * Dispatcher must be created by application and placed into {@link javax.servlet.ServletContext}
017     * using {@code dispatcherKeyInServletContext} parameter of {@link HandlersDispatcherServlet}
018     * Handler patterns are registered using <a href="https://github.com/tony19/named-regexp">named-regexes</a>.
019     *
020     * @author alexey
021     *         Date: 4/4/12
022     * @see HandlersDispatcherServlet
023     */
024    public class HandlersDispatcher {
025        private final List<HandlersMappingEntry<?>> registryGet;
026        private final List<HandlersMappingEntry<?>> registryPost;
027        private final List<HandlersMappingEntry<?>> registryPut;
028        private final NotFoundHandler notFoundHandler;
029        private final ExceptionHandler exceptionHandler;
030    
031        private HandlersDispatcher(List<HandlersMappingEntry<?>> registryGet, List<HandlersMappingEntry<?>> registryPost, List<HandlersMappingEntry<?>> registryPut,
032                                   NotFoundHandler notFoundHandler, ExceptionHandler exceptionHandler) {
033            this.registryGet = registryGet;
034            this.registryPost = registryPost;
035            this.registryPut = registryPut;
036            this.notFoundHandler = notFoundHandler;
037            this.exceptionHandler = exceptionHandler;
038        }
039    
040        /**
041         * Returns builder for {@code HandlersDispatcher}
042         *
043         * @return builder instance
044         */
045        public static Builder builder() {
046            return new Builder();
047        }
048    
049        /**
050         * Main dispatch method. Chooses registry using request method, than match {@code request.getPathInfo()}
051         * against patterns in registry. First matched handler will be used.
052         *
053         * @param req  request
054         * @param resp response
055         */
056        @SuppressWarnings("unchecked") // generic entries
057        public void dispatch(HttpServletRequest req, HttpServletResponse resp) {
058            try {
059                final List<HandlersMappingEntry<?>> mapping;
060                if("GET".equals(req.getMethod())) mapping = registryGet;
061                else if("POST".equals(req.getMethod())) mapping = registryPost;
062                else if("PUT".equals(req.getMethod())) mapping = registryPut;
063                    // cannot happen, methods are filtered in servlet
064                else throw new IllegalArgumentException("Unsupported HTTP method: '" + req.getMethod() + "'");
065                for(HandlersMappingEntry en : mapping) {
066                    NamedMatcher matcher = en.matcher(req.getPathInfo());
067                    if(matcher.matches()) {
068                        en.getTransportHandler().handle(en.getRestHandlerClass(), req, resp, matcher.namedGroups());
069                        return;
070                    }
071                }
072                notFoundHandler.handle(this, req, resp);
073            } catch(Exception e) {
074                exceptionHandler.handle(e, req, resp);
075            }
076        }
077    
078        /**
079         * Returns registry for {@code GET} method
080         *
081         * @return registry for {@code GET} method
082         */
083        public List<HandlersMappingEntry<?>> getRegistryGet() {
084            return unmodifiableList(registryGet);
085        }
086    
087        /**
088         * Returns registry for {@code POST} method
089         *
090         * @return registry for {@code POST} method
091         */
092        public List<HandlersMappingEntry<?>> getRegistryPost() {
093            return unmodifiableList(registryPost);
094        }
095    
096        /**
097         * Returns registry for {@code PUT} method
098         *
099         * @return registry for {@code PUT} method
100         */
101        public List<HandlersMappingEntry<?>> getRegistryPut() {
102            return unmodifiableList(registryPut);
103        }
104    
105        @Override
106        public String toString() {
107            final StringBuilder sb = new StringBuilder();
108            sb.append("HandlersDispatcher");
109            sb.append("{registryGet=").append(registryGet);
110            sb.append(", registryPost=").append(registryPost);
111            sb.append(", registryPut=").append(registryPut);
112            sb.append(", notFoundHandler=").append(notFoundHandler);
113            sb.append(", exceptionHandler=").append(exceptionHandler);
114            sb.append('}');
115            return sb.toString();
116        }
117    
118        /**
119         * Builder class for {@link HandlersDispatcher}
120         */
121        public static class Builder {
122            private final List<HandlersMappingEntry<?>> mappingGet = new ArrayList<HandlersMappingEntry<?>>();
123            private final List<HandlersMappingEntry<?>> mappingPost = new ArrayList<HandlersMappingEntry<?>>();
124            private final List<HandlersMappingEntry<?>> mappingPut = new ArrayList<HandlersMappingEntry<?>>();
125            private NotFoundHandler notFoundHandler;
126            private ExceptionHandler exceptionHandler;
127    
128            /**
129             * Constructor
130             */
131            public Builder() {
132            }
133    
134            /**
135             * Registers handler for {@code GET} requests
136             *
137             * @param pattern named regex pattern
138             * @param transportHandler transport handler instance
139             * @param appHandlerClass app handler class
140             * @param <T> app handler type
141             * @return builder itself
142             */
143            @SuppressWarnings("unchecked")
144            public <T extends RestHandler> Builder addGet(String pattern, TransportHandler<T> transportHandler,
145                                                          Class<? extends T> appHandlerClass) {
146                mappingGet.add(new HandlersMappingEntry<T>(pattern, transportHandler, (Class) appHandlerClass));
147                return this;
148            }
149    
150            /**
151             * Registers handler for {@code POST} requests
152             *
153             * @param pattern named regex pattern
154             * @param transportHandler transport handler instance
155             * @param appHandlerClass app handler class
156             * @param <T> app handler type
157             * @return builder itself
158             */
159            @SuppressWarnings("unchecked")
160            public <T extends RestHandler> Builder addPost(String pattern, TransportHandler<T> transportHandler,
161                                                           Class<? extends T> appHandlerClass) {
162                mappingPost.add(new HandlersMappingEntry<T>(pattern, transportHandler, (Class) appHandlerClass));
163                return this;
164            }
165    
166            /**
167             * Registers handler for {@code PUT} requests
168             *
169             * @param pattern named regex pattern
170             * @param transportHandler transport handler instance
171             * @param appHandlerClass app handler class
172             * @param <T> app handler type
173             * @return builder itself
174             */
175            @SuppressWarnings("unchecked")
176            public <T extends RestHandler> Builder addPut(String pattern, TransportHandler<T> transportHandler,
177                                                          Class<? extends T> appHandlerClass) {
178                mappingPut.add(new HandlersMappingEntry<T>(pattern, transportHandler, (Class) appHandlerClass));
179                return this;
180            }
181    
182            /**
183             * Adds all GET, PUT and DELETE handlers from specified builder to this builder.
184             * Not-found and exception handlers are ignored.
185             *
186             * @param otherBuilder builder to add handlers from
187             * @return builder itself
188             */
189            public Builder addHandlersFromBuilder(Builder otherBuilder) {
190                mappingGet.addAll(otherBuilder.mappingGet);
191                mappingPost.addAll(otherBuilder.mappingPost);
192                mappingPut.addAll(otherBuilder.mappingPut);
193                return this;
194            }
195    
196            /**
197             * Sets handler to use when no handler can be found in registry for provided request
198             *
199             * @param notFoundHandler not-found handler
200             * @return builder itself
201             */
202            public Builder setNotFound(NotFoundHandler notFoundHandler) {
203                this.notFoundHandler = notFoundHandler;
204                return this;
205            }
206    
207            /**
208             * Sets handler to use on exception happened during request processing
209             *
210             * @param exceptionHandler exception handler
211             * @return builder itself
212             */
213            public Builder setException(ExceptionHandler exceptionHandler) {
214                this.exceptionHandler = exceptionHandler;
215                return this;
216            }
217    
218            /**
219             * Builds dispatcher instance
220             *
221             * @return dispatcher instance
222             */
223            public HandlersDispatcher build() {
224                NotFoundHandler nfh = null != this.notFoundHandler ? this.notFoundHandler : new DefaultNotFoundHandler();
225                ExceptionHandler eh = null != this.exceptionHandler ? this.exceptionHandler : new DefaultExceptionHandler();
226                return new HandlersDispatcher(mappingGet, mappingPost, mappingPut, nfh, eh);
227            }
228        }
229    }
