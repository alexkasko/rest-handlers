001package com.alexkasko.rest.handlers;
002
003import javax.servlet.http.HttpServletRequest;
004import javax.servlet.http.HttpServletResponse;
005
006import static javax.servlet.http.HttpServletResponse.SC_NOT_FOUND;
007
008/**
009 * Default implementation of {@link NotFoundHandler}, returns plain-text 404 with
010 * registered handlers list.
011 *
012 * @author alexkasko
013 * Date: 11/14/12
014 */
015public class DefaultNotFoundHandler implements NotFoundHandler {
016    /**
017     * {@inheritDoc}
018     */
019    @Override
020    public void handle(HandlersDispatcher dispatcher, HttpServletRequest request, HttpServletResponse response) throws Exception {
021        response.setContentType("text/plain");
022        response.setStatus(SC_NOT_FOUND);
023        response.getWriter().write("Cannot find handler for request, path: '" + request.getPathInfo() + "'\n");
024        response.getWriter().write("Registered handlers:\n");
025        for(HandlersMappingEntry en : dispatcher.getRegistryGet()) {
026            response.getWriter().write("GET: ");
027            response.getWriter().write(en.getPattern());
028            response.getWriter().write("\n");
029        }
030        for(HandlersMappingEntry en : dispatcher.getRegistryPost()) {
031            response.getWriter().write("POST: ");
032            response.getWriter().write(en.getPattern());
033            response.getWriter().write("\n");
034        }
035        for(HandlersMappingEntry en : dispatcher.getRegistryPut()) {
036            response.getWriter().write("PUT: ");
037            response.getWriter().write(en.getPattern());
038            response.getWriter().write("\n");
039        }
040        for(HandlersMappingEntry en : dispatcher.getRegistryDelete()) {
041            response.getWriter().write("DELETE: ");
042            response.getWriter().write(en.getPattern());
043            response.getWriter().write("\n");
044        }
045    }
046}
