001    package com.alexkasko.rest.handlers.json;
002    
003    import com.alexkasko.rest.handlers.TransportHandler;
004    import com.google.gson.Gson;
005    
006    import javax.servlet.http.HttpServletRequest;
007    import javax.servlet.http.HttpServletResponse;
008    import java.util.Map;
009    
010    /**
011     * TransportHandler implementation for JSON requests and responses
012     *
013     * @author alexkasko
014     * Date: 11/14/12
015     */
016    public class JsonTransportHandler implements TransportHandler<JsonHandler> {
017        // may be preconfigured
018        private static final Gson gson = new Gson();
019    
020        /**
021         * Instantiate handler class, got input class form it, parses input object from request,
022         * gives it to handlers, writes results to response as JSON
023         *
024         * @param handlerClass application handler class
025         * @param request request
026         * @param response response
027         * @param urlParams named parameters from request path
028         * @throws Exception on any app exception
029         */
030        @Override
031        @SuppressWarnings("unchecked")
032        public void handle(Class<? extends JsonHandler> handlerClass, HttpServletRequest request,
033                           HttpServletResponse response, Map<String, String> urlParams) throws Exception {
034            // set proper content-type
035            response.setContentType("application/json");
036            // obtain handler instance, may be singleton, from DI etc
037            JsonHandler ha = handlerClass.newInstance();
038            // parse input object from request body
039            Object in = gson.fromJson(request.getReader(), ha.inputClass());
040            // fire handler
041            Object out = ha.handle(in);
042            // write results to client
043            gson.toJson(out, response.getWriter());
044        }
045    }
