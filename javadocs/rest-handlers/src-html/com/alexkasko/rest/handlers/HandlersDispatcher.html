001package com.alexkasko.rest.handlers;
002
003import com.google.code.regexp.NamedMatcher;
004
005import javax.servlet.http.HttpServletRequest;
006import javax.servlet.http.HttpServletResponse;
007import java.util.ArrayList;
008import java.util.List;
009
010import static java.util.Collections.unmodifiableList;
011
012/**
013 * Dispatches request using handlers registry matching with regular expression.
014 * Contains separate registries for {@code GET}, {@code POST} and {@code PUT} requests.
015 * Registries are ordered in handlers registration order. First matched handler will be used.
016 * Dispatcher must be created by application and placed into {@link javax.servlet.ServletContext}
017 * using {@code dispatcherKeyInServletContext} parameter of {@link HandlersDispatcherServlet}
018 * Handler patterns are registered using <a href="https://github.com/tony19/named-regexp">named-regexes</a>.
019 *
020 * @author alexey
021 *         Date: 4/4/12
022 * @see HandlersDispatcherServlet
023 */
024public class HandlersDispatcher {
025    private final List<HandlersMappingEntry<?>> registryGet;
026    private final List<HandlersMappingEntry<?>> registryPost;
027    private final List<HandlersMappingEntry<?>> registryPut;
028    private final List<HandlersMappingEntry<?>> registryDelete;
029    private final NotFoundHandler notFoundHandler;
030    private final ExceptionHandler exceptionHandler;
031
032    private HandlersDispatcher(List<HandlersMappingEntry<?>> registryGet, List<HandlersMappingEntry<?>> registryPost,
033                               List<HandlersMappingEntry<?>> registryPut, List<HandlersMappingEntry<?>> registryDelete,
034                               NotFoundHandler notFoundHandler, ExceptionHandler exceptionHandler) {
035        this.registryGet = registryGet;
036        this.registryPost = registryPost;
037        this.registryPut = registryPut;
038        this.registryDelete = registryDelete;
039        this.notFoundHandler = notFoundHandler;
040        this.exceptionHandler = exceptionHandler;
041    }
042
043    /**
044     * Returns builder for {@code HandlersDispatcher}
045     *
046     * @return builder instance
047     */
048    public static Builder builder() {
049        return new Builder();
050    }
051
052    /**
053     * Main dispatch method. Chooses registry using request method, than match {@code request.getPathInfo()}
054     * against patterns in registry. First matched handler will be used.
055     *
056     * @param req  request
057     * @param resp response
058     */
059    @SuppressWarnings("unchecked") // generic entries
060    public void dispatch(HttpServletRequest req, HttpServletResponse resp) {
061        try {
062            final List<HandlersMappingEntry<?>> mapping;
063            if("GET".equals(req.getMethod())) mapping = registryGet;
064            else if("POST".equals(req.getMethod())) mapping = registryPost;
065            else if("PUT".equals(req.getMethod())) mapping = registryPut;
066            else if("DELETE".equals(req.getMethod())) mapping = registryDelete;
067                // cannot happen, methods are filtered in servlet
068            else throw new IllegalArgumentException("Unsupported HTTP method: '" + req.getMethod() + "'");
069            for(HandlersMappingEntry en : mapping) {
070                NamedMatcher matcher = en.matcher(req.getPathInfo());
071                if(matcher.matches()) {
072                    en.getTransportHandler().handle(en.getRestHandlerClass(), req, resp, matcher.namedGroups());
073                    return;
074                }
075            }
076            notFoundHandler.handle(this, req, resp);
077        } catch(Exception e) {
078            exceptionHandler.handle(e, req, resp);
079        }
080    }
081
082    /**
083     * Returns registry for {@code GET} method
084     *
085     * @return registry for {@code GET} method
086     */
087    public List<HandlersMappingEntry<?>> getRegistryGet() {
088        return unmodifiableList(registryGet);
089    }
090
091    /**
092     * Returns registry for {@code POST} method
093     *
094     * @return registry for {@code POST} method
095     */
096    public List<HandlersMappingEntry<?>> getRegistryPost() {
097        return unmodifiableList(registryPost);
098    }
099
100    /**
101     * Returns registry for {@code PUT} method
102     *
103     * @return registry for {@code PUT} method
104     */
105    public List<HandlersMappingEntry<?>> getRegistryPut() {
106        return unmodifiableList(registryPut);
107    }
108
109    /**
110     * Returns registry for {@code DELETE} method
111     *
112     * @return registry for {@code DELETE} method
113     */
114    public List<HandlersMappingEntry<?>> getRegistryDelete() {
115        return unmodifiableList(registryDelete);
116    }
117
118    @Override
119    public String toString() {
120        final StringBuilder sb = new StringBuilder();
121        sb.append("HandlersDispatcher");
122        sb.append("{registryGet=").append(registryGet);
123        sb.append(", registryPost=").append(registryPost);
124        sb.append(", registryPut=").append(registryPut);
125        sb.append(", registryDelete=").append(registryDelete);
126        sb.append(", notFoundHandler=").append(notFoundHandler);
127        sb.append(", exceptionHandler=").append(exceptionHandler);
128        sb.append('}');
129        return sb.toString();
130    }
131
132    /**
133     * Builder class for {@link HandlersDispatcher}
134     */
135    public static class Builder {
136        private final List<HandlersMappingEntry<?>> mappingGet = new ArrayList<HandlersMappingEntry<?>>();
137        private final List<HandlersMappingEntry<?>> mappingPost = new ArrayList<HandlersMappingEntry<?>>();
138        private final List<HandlersMappingEntry<?>> mappingPut = new ArrayList<HandlersMappingEntry<?>>();
139        private final List<HandlersMappingEntry<?>> mappingDelete = new ArrayList<HandlersMappingEntry<?>>();
140        private NotFoundHandler notFoundHandler;
141        private ExceptionHandler exceptionHandler;
142
143        /**
144         * Constructor
145         */
146        public Builder() {
147        }
148
149        /**
150         * Registers handler for {@code GET} requests
151         *
152         * @param pattern named regex pattern
153         * @param transportHandler transport handler instance
154         * @param appHandlerClass app handler class
155         * @param <T> app handler type
156         * @return builder itself
157         */
158        @SuppressWarnings("unchecked")
159        public <T extends RestHandler> Builder addGet(String pattern, TransportHandler<T> transportHandler,
160                                                      Class<? extends T> appHandlerClass) {
161            mappingGet.add(new HandlersMappingEntry<T>(pattern, transportHandler, (Class) appHandlerClass));
162            return this;
163        }
164
165        /**
166         * Registers handler for {@code POST} requests
167         *
168         * @param pattern named regex pattern
169         * @param transportHandler transport handler instance
170         * @param appHandlerClass app handler class
171         * @param <T> app handler type
172         * @return builder itself
173         */
174        @SuppressWarnings("unchecked")
175        public <T extends RestHandler> Builder addPost(String pattern, TransportHandler<T> transportHandler,
176                                                       Class<? extends T> appHandlerClass) {
177            mappingPost.add(new HandlersMappingEntry<T>(pattern, transportHandler, (Class) appHandlerClass));
178            return this;
179        }
180
181        /**
182         * Registers handler for {@code PUT} requests
183         *
184         * @param pattern named regex pattern
185         * @param transportHandler transport handler instance
186         * @param appHandlerClass app handler class
187         * @param <T> app handler type
188         * @return builder itself
189         */
190        @SuppressWarnings("unchecked")
191        public <T extends RestHandler> Builder addPut(String pattern, TransportHandler<T> transportHandler,
192                                                      Class<? extends T> appHandlerClass) {
193            mappingPut.add(new HandlersMappingEntry<T>(pattern, transportHandler, (Class) appHandlerClass));
194            return this;
195        }
196
197        /**
198         * Registers handler for {@code PUT} requests
199         *
200         * @param pattern named regex pattern
201         * @param transportHandler transport handler instance
202         * @param appHandlerClass app handler class
203         * @param <T> app handler type
204         * @return builder itself
205         */
206        @SuppressWarnings("unchecked")
207        public <T extends RestHandler> Builder addDelete(String pattern, TransportHandler<T> transportHandler,
208                                                      Class<? extends T> appHandlerClass) {
209            mappingDelete.add(new HandlersMappingEntry<T>(pattern, transportHandler, (Class) appHandlerClass));
210            return this;
211        }
212
213
214        /**
215         * Adds all GET, PUT and DELETE handlers from specified list to this builder
216         * prepending specified prefix to all patterns from the list.
217         *
218         * @param list list of handlers mapping
219         * @return builder itself
220         */
221        @SuppressWarnings("unchecked")
222        public Builder addList(String prefix, HandlersSubmapping list) {
223            for(HandlersMappingEntry<?> en : list.getMappingGet()) {
224                mappingGet.add(new HandlersMappingEntry(prefix, en));
225            }
226            for(HandlersMappingEntry<?> en : list.getMappingPost()) {
227                mappingPost.add(new HandlersMappingEntry(prefix, en));
228            }
229            for(HandlersMappingEntry<?> en : list.getMappingPut()) {
230                mappingPut.add(new HandlersMappingEntry(prefix, en));
231            }
232            for(HandlersMappingEntry<?> en : list.getMappingDelete()) {
233                mappingDelete.add(new HandlersMappingEntry(prefix, en));
234            }
235            return this;
236        }
237
238        /**
239         * Sets handler to use when no handler can be found in registry for provided request
240         *
241         * @param notFoundHandler not-found handler
242         * @return builder itself
243         */
244        public Builder setNotFound(NotFoundHandler notFoundHandler) {
245            this.notFoundHandler = notFoundHandler;
246            return this;
247        }
248
249        /**
250         * Sets handler to use on exception happened during request processing
251         *
252         * @param exceptionHandler exception handler
253         * @return builder itself
254         */
255        public Builder setException(ExceptionHandler exceptionHandler) {
256            this.exceptionHandler = exceptionHandler;
257            return this;
258        }
259
260        /**
261         * Builds dispatcher instance
262         *
263         * @return dispatcher instance
264         */
265        public HandlersDispatcher build() {
266            NotFoundHandler nfh = null != this.notFoundHandler ? this.notFoundHandler : new DefaultNotFoundHandler();
267            ExceptionHandler eh = null != this.exceptionHandler ? this.exceptionHandler : new DefaultExceptionHandler();
268            return new HandlersDispatcher(mappingGet, mappingPost, mappingPut, mappingDelete, nfh, eh);
269        }
270    }
271}
