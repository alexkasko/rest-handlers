001package com.alexkasko.rest.handlers;
002
003import java.util.ArrayList;
004import java.util.List;
005
006/**
007 * List of handlers mappings for submodules configuration
008 *
009 * @author alexkasko
010 * Date: 7/18/13
011 */
012public class HandlersSubmapping {
013    private final List<HandlersMappingEntry<? extends RestHandler>> mappingGet = new ArrayList<HandlersMappingEntry<? extends RestHandler>>();
014    private final List<HandlersMappingEntry<? extends RestHandler>> mappingPost = new ArrayList<HandlersMappingEntry<? extends RestHandler>>();
015    private final List<HandlersMappingEntry<? extends RestHandler>> mappingPut = new ArrayList<HandlersMappingEntry<? extends RestHandler>>();
016    private final List<HandlersMappingEntry<? extends RestHandler>> mappingDelete = new ArrayList<HandlersMappingEntry<? extends RestHandler>>();
017
018    /**
019     * Registers handler for {@code GET} requests
020     *
021     * @param pattern          named regex pattern
022     * @param transportHandler transport handler instance
023     * @param appHandlerClass  app handler class
024     * @param <T>              app handler type
025     * @return builder itself
026     */
027    @SuppressWarnings("unchecked")
028    public <T extends RestHandler> HandlersSubmapping addGet(String pattern, TransportHandler<T> transportHandler,
029                                                              Class<? extends T> appHandlerClass) {
030        mappingGet.add(new HandlersMappingEntry<T>(pattern, transportHandler, (Class) appHandlerClass));
031        return this;
032    }
033
034    /**
035     * Registers handler for {@code POST} requests
036     *
037     * @param pattern          named regex pattern
038     * @param transportHandler transport handler instance
039     * @param appHandlerClass  app handler class
040     * @param <T>              app handler type
041     * @return builder itself
042     */
043    @SuppressWarnings("unchecked")
044    public <T extends RestHandler> HandlersSubmapping addPost(String pattern, TransportHandler<T> transportHandler,
045                                                               Class<? extends T> appHandlerClass) {
046        mappingPost.add(new HandlersMappingEntry<T>(pattern, transportHandler, (Class) appHandlerClass));
047        return this;
048    }
049
050    /**
051     * Registers handler for {@code PUT} requests
052     *
053     * @param pattern          named regex pattern
054     * @param transportHandler transport handler instance
055     * @param appHandlerClass  app handler class
056     * @param <T>              app handler type
057     * @return builder itself
058     */
059    @SuppressWarnings("unchecked")
060    public <T extends RestHandler> HandlersSubmapping addPut(String pattern, TransportHandler<T> transportHandler,
061                                                              Class<? extends T> appHandlerClass) {
062        mappingPut.add(new HandlersMappingEntry<T>(pattern, transportHandler, (Class) appHandlerClass));
063        return this;
064    }
065
066    /**
067     * Registers handler for {@code DELETE} requests
068     *
069     * @param pattern          named regex pattern
070     * @param transportHandler transport handler instance
071     * @param appHandlerClass  app handler class
072     * @param <T>              app handler type
073     * @return builder itself
074     */
075    @SuppressWarnings("unchecked")
076    public <T extends RestHandler> HandlersSubmapping addDelete(String pattern, TransportHandler<T> transportHandler,
077                                                              Class<? extends T> appHandlerClass) {
078        mappingDelete.add(new HandlersMappingEntry<T>(pattern, transportHandler, (Class) appHandlerClass));
079        return this;
080    }
081
082
083    List<HandlersMappingEntry<? extends RestHandler>> getMappingGet() {
084        return mappingGet;
085    }
086
087    List<HandlersMappingEntry<? extends RestHandler>> getMappingPost() {
088        return mappingPost;
089    }
090
091    List<HandlersMappingEntry<? extends RestHandler>> getMappingPut() {
092        return mappingPut;
093    }
094
095    List<HandlersMappingEntry<? extends RestHandler>> getMappingDelete() {
096        return mappingDelete;
097    }
098
099    /**
100     * {@inheritDoc}
101     */
102    @Override
103    public String toString() {
104        final StringBuilder sb = new StringBuilder();
105        sb.append("HandlersMappingList");
106        sb.append("{mappingGet=").append(mappingGet);
107        sb.append(", mappingPost=").append(mappingPost);
108        sb.append(", mappingPut=").append(mappingPut);
109        sb.append(", mappingDelete=").append(mappingDelete);
110        sb.append('}');
111        return sb.toString();
112    }
113}
