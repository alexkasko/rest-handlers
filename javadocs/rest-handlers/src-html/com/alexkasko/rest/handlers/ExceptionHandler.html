001    package com.alexkasko.rest.handlers;
002    
003    import javax.servlet.http.HttpServletRequest;
004    import javax.servlet.http.HttpServletResponse;
005    
006    /**
007     * Used on exception happened during request processing
008     *
009     * @author alexkasko
010     * Date: 11/14/12
011     */
012    public interface ExceptionHandler {
013        /**
014         * Implementation may report exception to client.
015         * Must not throw runtime exception.
016         *
017         * @param e exception happened during request processing
018         * @param request request
019         * @param response response
020         */
021        void handle(Exception e, HttpServletRequest request, HttpServletResponse response);
022    }
