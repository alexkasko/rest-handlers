001package com.alexkasko.rest.handlers.json;
002
003/**
004 * {@link JsonHandler} example implementation, return input message untouched
005 *
006 * @author alexkasko
007 * Date: 11/14/12
008 */
009public class EchoHandler implements JsonHandler<InputMessage, OutputMessage> {
010
011    /**
012     * Says hello back untouched
013     *
014     * @param input input object parsed from json
015     * @return input message untouched
016     */
017    @Override
018    public OutputMessage handle(InputMessage input) {
019        return new OutputMessage(input.getText());
020    }
021
022    /**
023     * Returns {@link InputMessage} class to be instantiated by transport handler
024     *
025     * @return {@link InputMessage} class
026     */
027    @Override
028    public Class<InputMessage> inputClass() {
029        return InputMessage.class;
030    }
031}
